24         private void LoadLanguageResources(int culture = 2052)
25         {
26             string fileName = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, string.Format("{0}.resources", culture));
27             ResourceReader<string> reader = new StringResourceReader(fileName);
28             reader.Read();
29             this.Text = reader.Resources["FormTitle"].Data;
30             this.CtrlSimplifiedChineseButton.Text = reader.Resources["Button2052"].Data;
31             this.CtrlEnglishButton.Text = reader.Resources["Button1033"].Data;
32             this.CtrlExitApplicationButton.Text = reader.Resources["ButtonClose"].Data;
33         }
34         #endregion
